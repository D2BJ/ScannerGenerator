$DIGIT    		[0-4 7]
$NON-ZERO    [^0]  IN  $DIGIT
$SMALLCASE   	[a-z]
$LETTER      	[A-Ea-z]
$UPPERCASE  [^a-z] IN $LETTER         
%%  Token definitions will be contained in this section using regexes – examples below
$IDENTIFIER     	$SMALLCASE ($LETTER | $DIGIT)*
